1. ¿Cuántos constructores tiene el objeto predefinido Date()? Explícalos mediante ejemplos.

El objeto Date() tiene 4 constructores.

  1. new Date()
    • Crea un nuevo objeto Date() con la fecha y hora actual.
    let d = new Date();
    document.open();
    document.write(d);
    document.close();
  2. new Date(milisegundos)
    • Crea un nuevo objeto Date() cómo un tiempo de 0 más el número, en milisegundos.
    let d = new Date(86400000);
    document.open();
    document.write(d);
    document.close();
  3. new Date(dateString) creates a new date object from the specified date and time:
    • Crea un nuevo objeto Date() con la fecha y hora espeficificada.
    let d = new Date("October 13, 2014 11:13:00");
    document.open();
    document.write(d);
    document.close();
  4. new Date(año, mes, días, horas, minutos, segundos, milisegundos)
    • Crea un nuevo objeto Date() con la fecha y hora especificadas.
    let d = new Date(99,5,24,11,33,30,0);
    document.open();
    document.write(d);
    document.close();
2. Indica la utilidad de Date.now();
  • Date.now()

5. Crea el método incrementaDias() que incremente/decremente los días indicados.

Admitirá como primer argumento un entero positivo/negativo que reperesente el número de días. El resto de argumentos representarán una fecha, similar a los argumentos del constructor Date().

let incrementaDias = function(cantidad, fecha) {
 let dia = fecha.getDay() + cantidad;
 fecha.setDate(dia);
 return fecha;
}

6. Implementa el método diaDeLaSemana() que devuelva 'lunes, martes... domingo' del día actual o de la fecha indicada.

let diaDeLaSemana = function(fecha) {
 if(fecha == null){
  fecha = new Date();
 }

 switch(fecha.getDay()) {
  case 0:
   return "Domingo";
   break;
  case 1:
   return "Lunes";
   break;
  case 2:
   return "Martes";
   break;

   break;
  case 4:

   break;

   break;
  case 6:
   return "Sábado";
   break;
 }
}

9. Implementa el método calcularHastaFinDeAnno() que devuelva los días que quedan hasta fin de año
let calcularHastaFinDeAnno = function() {
 let hoy = new Date();
 let anoNuevo = new Date(hoy.getFullYear(), 11, 31);
 anoNuevo.setDate(anoNuevo.getDate() + 1);

 let diasDif = anoNuevo.getTime() - hoy.getTime();
 let dias = Math.round(diasDif/(1000 * 60 * 60 * 24));

 return dias;
}
